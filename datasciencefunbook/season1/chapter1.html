import pandas as pd
import numpy as np

df = pd.read_csv("customer_satisfaction_dirty.csv")
print(df.info())
print(df.describe(include='all'))

df['rating'] = df['rating'].replace({'bad': 1, 'average': 3, 'good': 5}).astype(float)
df['rating'].fillna(df['rating'].median(), inplace=True)

df['state'] = df['state'].str.upper().str.strip().replace({
    'CALIFORNIA': 'CA', 'CALIF.': 'CA', 'CA ': 'CA'
})

df['response_date'] = pd.to_datetime(df['response_date'], errors='coerce')

q1 = df['time_on_site'].quantile(0.25)
q3 = df['time_on_site'].quantile(0.75)
iqr = q3 - q1
lower = q1 - 1.5 * iqr
upper = q3 + 1.5 * iqr
df = df[(df['time_on_site'] >= lower) & (df['time_on_site'] <= upper)]

df = df[(df['age'] >= 0) & (df['age'] <= 120)]
